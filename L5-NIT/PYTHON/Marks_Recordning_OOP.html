# ==============================
# MARKS RECORDING SYSTEM
# USING OOP AND LIST
# ==============================

class Students:

    def __init__(self, name, maths, kiny, kisw):
        self.name = name
        self.maths = maths
        self.kiny = kiny
        self.kisw = kisw

    def cal_total(self):
        total = self.maths + self.kiny + self.kisw
        return total

    def cal_average(self):
        total = self.cal_total()
        average = total / 3
        return average

    def display_results(self):
        print("Student Name:", self.name)
        print("Maths:", self.maths)
        print("Kinyarwanda:", self.kiny)
        print("Kiswahili:", self.kisw)
        print("Total Marks:", self.cal_total())
        print("Average:", self.cal_average())
        print("------------------------")


students_list = []

for i in range(2):

    name = input("Enter the names: ")
    maths = float(input("How many marks for Maths: "))
    kiny = float(input("How many marks for Kinyarwanda: "))
    kisw = float(input("How many marks for Kiswahili: "))

    student = Students(name, maths, kiny, kisw)
    students_list.append(student)

for student in students_list:
    student.display_results()
