Python OOP Modules System


class Modules:
    # Constructor
    def __init__(self, mod_code, mod_name):
        # Private attributes (Encapsulation)
        self.__modCode = mod_code
        self.__modName = mod_name

    # Getter
    def get_modData(self):
        return self.__modCode, self.__modName

    # Setter
    def set_modData(self, mod_code, mod_name):
        self.__modCode = mod_code
        self.__modName = mod_name

    # Add module
    def addModule(self):
        modCode = input("Enter module code: ")
        modName = input("Enter module name: ")

        module = {
            'modCode': modCode,
            'modName': modName
        }

        return module

    # Display module
    def displayModule(self, module):
        print(f"| {module['modCode']:<15} | {module['modName']:<25} |")


# List to store modules
modules_list = []

# Menu system
while True:
    print("\n================ MENU ================")
    print("1. Add Module")
    print("2. Display Modules")
    print("3. Exit")
    print("======================================")

    choice = input("Enter your choice: ")

    if choice == '1':
        module = Modules("", "")
        mod_data = module.addModule()
        modules_list.append(mod_data)
        print("Module added successfully!")

    elif choice == '2':
        if len(modules_list) == 0:
            print("No modules available.")
        else:
            # Table Header
            print("\n+-----------------+---------------------------+")
            print(f"| {'Module Code':<15} | {'Module Name':<25} |")
            print("+-----------------+---------------------------+")

            # Table Data
            for mod in modules_list:
                module = Modules("", "")
                module.displayModule(mod)

            print("+-----------------+---------------------------+")

    elif choice == '3':
        print("Exiting the program.")
        break

    else:
        print("Invalid choice. Please try again.")
