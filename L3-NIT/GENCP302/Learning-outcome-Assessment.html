PART II: PROGRAM DEBUGGING (20 Marks)
Question 31
#include 

int main()
{
    printf("Hello World")
    return 0;
}
a) Error Type: Syntax Error
b) Reason: Missing semicolon (;) after printf statement.
c) Correct Program: 
#include 

int main()
{
    printf("Hello World");
    return 0;
}
Question 32
#include 

int main()
{
    int a = 10, b = 5;
    int sum;

    sum = a - b;

    printf("Sum = %d", sum);

    return 0;
}
a) Error Type: Logical Error
b) Reason: The program is supposed to calculate a sum but uses subtraction instead of addition.
c) Correct Program: 
#include 

int main()
{
    int a = 10, b = 5;
    int sum;

    sum = a + b;

    printf("Sum = %d", sum);

    return 0;
}
Question 33
#include 

int main()
{
    int a = 10;
    int b = 0;

    printf("%d", a / b);

    return 0;
}
a) Error Type: Runtime Error
b) Reason: Division by zero is not allowed and causes program failure during execution.
c) Correct Program: 
#include 

int main()
{
    int a = 10;
    int b = 2;

    printf("%d", a / b);

    return 0;
}
