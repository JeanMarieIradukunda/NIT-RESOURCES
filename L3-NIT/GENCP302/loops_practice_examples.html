🟢 1️⃣ Attendance Marking System

Question: Write a C program for 3 classes with 5 students each. Input attendance (1=Present, 0=Absent). Display total present students per class. Use nested loops. 
#include <stdio.h>

int main()
{
    int class, student;
    int attendance;
    int totalPresent;

    for(class = 1; class <= 3; class++)
    {
        totalPresent = 0;

        for(student = 1; student <= 5; student++)
        {
            printf("Enter attendance (1/0): ");
            scanf("%d", &attendance);

            if(attendance == 1)
            {
                totalPresent++;
            }
        }

        printf("Total Present in Class %d = %d\n",
               class, totalPresent);
    }

    return 0;
}

🟢 2️⃣ Student Marks Recording

Question: 5 students, 4 subjects each. Input marks, calculate total and average. Use nested loops. 
#include <stdio.h>

int main()
{
    int student, subject;
    float marks, total, average;

    for(student = 1; student <= 5; student++)
    {
        total = 0;

        for(subject = 1; subject <= 4; subject++)
        {
            scanf("%f", &marks);
            total += marks;
        }

        average = total / 4;

        printf("Total = %.2f\n", total);
        printf("Average = %.2f\n", average);
    }

    return 0;
}

🟢 3️⃣ Classroom Seating (Parts & Tables)

Question: 3 parts in a classroom. Each part has 7 tables. Number tables starting from 1. Use nested loops. 
#include <stdio.h>

int main()
{
    int part, table;
    int tableNumber = 1;

    for(part = 1; part <= 3; part++)
    {
        printf("\n--- Part %d ---\n", part);

        for(table = 1; table <= 7; table++)
        {
            printf("Table %d\t", tableNumber);
            tableNumber++;
        }

        printf("\n");
    }

    return 0;
}

🟢 4️⃣ School Timetable Generator

Question: 5 days, 8 periods per day. Input subject names and display timetable. Use nested loops. 
#include <stdio.h>

int main()
{
    int day, period;
    char subject[30];

    for(day = 1; day <= 5; day++)
    {
        printf("\nDay %d:\n", day);

        for(period = 1; period <= 8; period++)
        {
            scanf("%s", subject);
            printf("Period %d: %s\n",
                   period, subject);
        }
    }

    return 0;
}
